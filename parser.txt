LEX Program (parser.l)
%{
#include "y.tab.h"
#include <stdlib.h>
%}

%%

[0-9]+      { yylval = atoi(yytext); return NUM; }
[ \t]       ;   // ignore spaces
\n          { return 0; }
.           { return yytext[0]; }

%%

int yywrap() {
    return 1;
}

YACC Program (parser.y)
%{
#include <stdio.h>
%}

%token NUM

%%

S : E { printf("\nResult = %d\n", $1); }
  ;

E : E '+' NUM { $$ = $1 + $3; }
  | NUM       { $$ = $1; }
  ;

%%

int main() {
    printf("Enter expression:\n");
    yyparse();
    return 0;
}

int yyerror(char *s) {
    printf("Error: %s\n", s);
    return 0;
}

How to Run
lex parser.l
yacc -d parser.y
gcc lex.yy.c y.tab.c -o parser
./parser
